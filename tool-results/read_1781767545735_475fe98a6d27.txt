   199тЖТ   199тЖТ  function getPasswordStrength(password: string): 'weak' | 'medium' | 'strong' {
   200тЖТ   200тЖТ    if (!password) return 'weak';
   201тЖТ   201тЖТ    let score = 0;
   202тЖТ   202тЖТ    if (password.length >= 6) score++;
   203тЖТ   203тЖТ    if (password.length >= 10) score++;
   204тЖТ   204тЖТ    if (/[A-Z]/.test(password)) score++;
   205тЖТ   205тЖТ    if (/[0-9]/.test(password)) score++;
   206тЖТ   206тЖТ    if (/[^A-Za-z0-9]/.test(password)) score++;
   207тЖТ   207тЖТ    if (score <= 2) return 'weak';
   208тЖТ   208тЖТ    if (score <= 3) return 'medium';
   209тЖТ   209тЖТ    return 'strong';
   210тЖТ   210тЖТ  }
